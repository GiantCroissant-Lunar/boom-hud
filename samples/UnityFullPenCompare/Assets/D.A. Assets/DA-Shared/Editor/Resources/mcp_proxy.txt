import sys
import json
import requests

MCP_SERVER_URL = "http://127.0.0.1:8673/"

def main():
    while True:
        try:
            line = sys.stdin.readline()
            if not line:
                break
            
            message = json.loads(line)
            
            try:
                response = requests.post(MCP_SERVER_URL, json=message)
                response.raise_for_status()
                
                if response.text:
                    print(response.text)
                    sys.stdout.flush()
                    
            except requests.exceptions.RequestException:
                pass 

        except json.JSONDecodeError:
            continue
        except KeyboardInterrupt:
            break
        except Exception:
            continue

if __name__ == "__main__":
    main()
